# Security headers pro guardiaokids.site
# ───────────────────────────────────────
#
# Adicionar ao .htaccess da raiz do WordPress (`~/domains/guardiaokids.site/public_html/.htaccess`)
# acima do bloco `# BEGIN WordPress`. NÃO sobrescrever — só prepend.
#
# Aplica em todas as respostas (PHP + assets estáticos).
# Cobre itens 15-20 da auditoria do site público.

<IfModule mod_headers.c>
    # HSTS — força HTTPS 1 ano + subdomínios; preload-ready
    Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains; preload"

    # CSP — restritivo o suficiente pra bloquear XSS, permissivo pro WP funcionar
    #  - Google Fonts (CSS + arquivos) liberado pra UI do plugin
    #  - 'unsafe-inline' em styles é exigência do WP block editor e do plugin atual
    #  - 'unsafe-inline' em scripts mantido por hora; remover quando block editor sair
    Header always set Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; font-src 'self' https://fonts.gstatic.com; img-src 'self' data: https:; connect-src 'self'; frame-ancestors 'self'; base-uri 'self'; form-action 'self'; upgrade-insecure-requests"

    # MIME sniffing protection
    Header always set X-Content-Type-Options "nosniff"

    # Clickjacking protection — SAMEORIGIN (não DENY pra permitir wp-admin iframe)
    Header always set X-Frame-Options "SAMEORIGIN"

    # Referrer — não vazar URLs internas
    Header always set Referrer-Policy "strict-origin-when-cross-origin"

    # Permissions-Policy — bloqueia features que não usamos
    # geolocation/camera mantidos sem restrição porque o PWA infantil consome
    Header always set Permissions-Policy "accelerometer=(), autoplay=(), browsing-topics=(), camera=(self), gyroscope=(), magnetometer=(), microphone=(), midi=(), payment=(), usb=(), interest-cohort=()"
</IfModule>
